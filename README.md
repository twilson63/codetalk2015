    var _ = require('underscore')

    function local(access_token) {
      return function (repos) {
        _(repos).each(upsert)
        return repos
      }
    }

    function upsert(repo) {
      db.upsert(repo.full_name, function (doc) {
        return _(doc).extend(repo)
      })
      .catch(function (err) {
        console.log(err)
      })
    }
